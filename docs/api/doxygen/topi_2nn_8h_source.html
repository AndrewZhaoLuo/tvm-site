153 inline tvm::te::Tensor pad(const tvm::te::Tensor& t, const tvm::Array<tvm::PrimExpr>& pad_before,
154  tvm::Array<tvm::PrimExpr> pad_after = tvm::Array<tvm::PrimExpr>(),
155  PrimExpr pad_value = PrimExpr(), std::string name = "T_pad",
156  std::string tag = kElementWise, std::string pad_mode = "constant",
157  const Array<PrimExpr>* dyn_output_shape = nullptr) {
158  if (pad_after.size() < pad_before.size()) {
159  for (size_t i = pad_after.size(); i < pad_before.size(); ++i) {
160  pad_after.push_back(pad_before[i]);
161  }
162  }
163 
164  arith::Analyzer analyzer;
165  CHECK_GE(pad_before.size(), 1);
166  CHECK_EQ(pad_before.size(), pad_after.size());
167  tvm::Array<tvm::PrimExpr> pad_before_int32;
168  tvm::Array<tvm::PrimExpr> pad_after_int32;
169 
170  for (const auto& ele : pad_before) {
171  pad_before_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
172  }
173  for (const auto& ele : pad_after) {
174  pad_after_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
175  }
176 
177  tvm::Array<tvm::PrimExpr> output_shape;
178  if (dyn_output_shape == nullptr) {
179  for (size_t i = 0; i < t->shape.size(); ++i) {
180  if (i >= pad_before.size()) {
181  output_shape.push_back(t->shape[i]);
182  } else {
183  output_shape.push_back(
184  analyzer.Simplify(t->shape[i] + pad_before_int32[i] + pad_after_int32[i]));
185  }
186  }
187  } else {
188  for (size_t i = 0; i < dyn_output_shape->size(); i++) {
189  output_shape.push_back((*dyn_output_shape)[i]);
190  }
191  }
192 
193  if (!pad_value.defined()) {
194  pad_value = tvm::tir::make_const(t->dtype, 0);
195  }
196 
197  auto l = [&](tvm::Array<tvm::tir::Var> ovars) {
198  tvm::Array<tvm::PrimExpr> indices;
199  tvm::Array<tvm::PrimExpr> sel;
200  tvm::Array<tvm::PrimExpr> pad_idx;
201  for (size_t i = 0; i < t->shape.size(); ++i) {
202  if (i >= pad_before_int32.size()) {
203  indices.push_back(ovars[i]);
204  continue;
205  }
206  if (!topi::detail::EqualCheck(pad_before_int32[i], 0)) {
207  sel.push_back(ovars[i] >= pad_before_int32[i]);
208  indices.push_back(ovars[i] - pad_before_int32[i]);
209  } else {
210  indices.push_back(ovars[i]);
211  }
212  if (!topi::detail::EqualCheck(pad_after_int32[i], 0)) {
213  sel.push_back(analyzer.Simplify(ovars[i] < pad_before_int32[i] + t->shape[i]));
214  }
215  if (pad_mode == "edge") {
216  pad_idx.push_back(
217  tvm::if_then_else(ovars[i] < pad_before[i], 0,
218  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
219  t->shape[i] - 1, ovars[i] - pad_before[i])));
220  } else if (pad_mode == "reflect") {
221  pad_idx.push_back(
222  tvm::if_then_else(ovars[i] < pad_before[i], pad_before[i] - ovars[i],
223  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
224  t->shape[i] * 2 - ovars[i] + pad_before[i] - 2,
225  ovars[i] - pad_before[i])));
226  }
227  }
228  if (sel.size() != 0) {
229  auto fand = [](PrimExpr a, PrimExpr b) { return a && b; };
230  if (pad_mode == "constant") {
231  return tvm::if_then_else(foldl(fand, const_true(1), sel), t(indices), pad_value);
232  } else if (pad_mode == "edge" || pad_mode == "reflect") {
233  return tvm::if_then_else(foldl(fand, const_true(1), sel), t(indices), t(pad_idx));
234  }
235  }
236  return t(indices);
237  };
238  return tvm::te::compute(output_shape, l, name, tag);
239 }

261 inline tvm::te::Tensor conv2d_nchw(const tvm::te::Tensor& I, const tvm::te::Tensor& W,
262  int pad_h = 0, int pad_w = 0, int stride_h = 1, int stride_w = 1,
263  std::string name = "T_conv2d_nchw",
264  std::string tag = kConv2dNCHW) {
265  CHECK_EQ(4, I->shape.size());
266  CHECK_EQ(4, W->shape.size());
267  auto pH = I->shape[2];
268  auto pW = I->shape[3];
269  tvm::Array<tvm::PrimExpr> output_shape{
270  I->shape[0], // B
271  W->shape[0], // O
272  indexdiv(I->shape[2] - W->shape[2] + 2 * pad_h, stride_h) + 1, // H
273  indexdiv(I->shape[3] - W->shape[3] + 2 * pad_w, stride_w) + 1 // W
274  };
275  auto i = tvm::te::reduce_axis(tvm::Range{0, I->shape[1]}, "i");
276  auto kh = tvm::te::reduce_axis(tvm::Range{0, W->shape[2]}, "kh");
277  auto kw = tvm::te::reduce_axis(tvm::Range{0, W->shape[3]}, "kw");
278  auto T =
279  (pad_h == 0 && pad_w == 0) ? I : pad(I, {tvm::PrimExpr(0), tvm::PrimExpr(0), pad_h, pad_w});
280  auto l = [&](tvm::tir::Var b, tvm::tir::Var o, tvm::tir::Var h, tvm::tir::Var w) {
281  return tvm::sum(T(b, i, stride_h * h + kh, stride_w * w + kw) * W(o, i, kh, kw), {i, kh, kw});
282  };
283  return tvm::te::compute(output_shape, l, name, tag);
284 }
