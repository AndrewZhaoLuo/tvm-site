47 inline tvm::te::Tensor broadcast_to(const tvm::te::Tensor& t,
48  const tvm::Array<tvm::PrimExpr>& output_shape,
49  std::string name = "T_broadcast_to",
50  std::string tag = kBroadcast) {
51  CHECK_GE(output_shape.size(), t->shape.size())
52  << "Not a broadcast, output dimensionality smaller than input.\noutput: " << output_shape
53  << "\nvs\ninput: " << t;
54  auto bh = detail::BroadcastShape(output_shape, t->shape);
55  CHECK_EQ(output_shape.size(), bh.common_shape.size());
56  for (size_t i = 0; i < output_shape.size(); ++i) {
57  CHECK(topi::detail::EqualCheck(output_shape[i], bh.common_shape[i]));
58  }
59  auto l = [&](tvm::Array<tvm::tir::Var> ovars) {
60  return t(detail::InputIndexFromBroadcast(ovars, t, bh.vars2, bh.all_vars));
61  };
62  return tvm::te::compute(tvm::Array<tvm::PrimExpr>(bh.common_shape.begin(), bh.common_shape.end()),
63  l, name, tag);
64 }
