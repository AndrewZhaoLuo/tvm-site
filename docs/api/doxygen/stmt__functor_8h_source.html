175 class TVM_DLL StmtMutator : protected StmtFunctor<Stmt(const Stmt&)> {
176  public:
185  Stmt operator()(Stmt stmt) {
186  allow_copy_on_write_ = true;
187  return VisitStmt(stmt);
188  }
189 
190  protected:
191  // We perform copy on write optimizations on the StmtMutator
192  // so that an unique copy of parent can be mutated inplace
193  // when some of its children changed.
194  // We only do such optimization for Stmt nests(instead of Exprs) for now
195  // as Stmt's parent state is more likely remain unchanged when one of
196  // its child block changes.
201  bool allow_copy_on_write_{false};
211  template <typename TNode>
212  ObjectPtr<TNode> CopyOnWrite(const TNode* node) {
213  if (allow_copy_on_write_) {
214  // return the old node.
215  return runtime::GetObjectPtr<TNode>(const_cast<TNode*>(node));
216  } else {
217  // Make a new copy of the node.
218  // need to rely on the default copy constructor
219  return runtime::make_object<TNode>(*node);
220  }
221  }
228  Stmt VisitStmt(const Stmt& stmt) override {
229  if (allow_copy_on_write_ && !stmt.unique()) {
230  allow_copy_on_write_ = false;
231  Stmt ret = StmtFunctor::VisitStmt(stmt);
232  allow_copy_on_write_ = true;
233  return ret;
234  } else {
235  return StmtFunctor::VisitStmt(stmt);
236  }
237  }
