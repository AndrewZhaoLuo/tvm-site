166 class TVM_DLL StmtMutator : protected StmtFunctor<Stmt(const Stmt&)> {
167  public:
176  Stmt operator()(Stmt stmt) {
177  allow_copy_on_write_ = true;
178  return VisitStmt(stmt);
179  }
180 
181  protected:
182  // We perform copy on write optimizations on the StmtMutator
183  // so that an unique copy of parent can be mutated inplace
184  // when some of its children changed.
185  // We only do such optimization for Stmt nests(instead of Exprs) for now
186  // as Stmt's parent state is more likely remain unchanged when one of
187  // its child block changes.
192  bool allow_copy_on_write_{false};
202  template <typename TNode>
203  ObjectPtr<TNode> CopyOnWrite(const TNode* node) {
204  if (allow_copy_on_write_) {
205  // return the old node.
206  return runtime::GetObjectPtr<TNode>(const_cast<TNode*>(node));
207  } else {
208  // Make a new copy of the node.
209  // need to rely on the default copy constructor
210  return runtime::make_object<TNode>(*node);
211  }
212  }
219  Stmt VisitStmt(const Stmt& stmt) override {
220  if (allow_copy_on_write_ && !stmt.unique()) {
221  allow_copy_on_write_ = false;
222  Stmt ret = StmtFunctor::VisitStmt(stmt);
223  allow_copy_on_write_ = true;
224  return ret;
225  } else {
226  return StmtFunctor::VisitStmt(stmt);
227  }
228  }
236  virtual PrimExpr VisitExpr(const PrimExpr& e) { return e; }
237  // statement visitor
