49 inline Tensor softmax(const Tensor& x, int axis = -1, std::string name = "tensor",
50  std::string tag = "softmax_output") {
51  auto input_shape = x->shape;
52  auto ndim = input_shape.size();
53  if (axis < 0) {
54  axis = ndim + axis;
55  }
56  CHECK_LT(axis, ndim) << "axis parameter should be less than input dim";
57 
58  auto k1 = tvm::te::reduce_axis(Range(0, input_shape[axis]), "k1");
59  auto k2 = tvm::te::reduce_axis(Range(0, input_shape[axis]), "k2");
60  auto reduced_shape = MakeReduceTargetShape({axis}, x, false, false);
61 
62  tvm::Map<String, ObjectRef> attrs;
63  attrs.Set("axis", Integer(axis));

65  auto insert_reduce_index = [axis, ndim](const Array<Var>& indices, const IterVar& reduce_index) {
66  Array<PrimExpr> eval_range;
67  int arg_counter = 0;
68  for (size_t i = 0; i < ndim; ++i) {
69  if (static_cast<int>(i) == axis)
70  eval_range.push_back(reduce_index);
71  else
72  eval_range.push_back(indices[arg_counter++]);
73  }
74  return eval_range;
75  };

77  auto get_non_reduce_indices = [axis, ndim](const Array<Var>& indices) {
78  Array<PrimExpr> non_reduce_indices;
79  for (size_t i = 0; i < ndim; ++i) {
80  if (static_cast<int>(i) != axis) non_reduce_indices.push_back(indices[i]);
81  }
82  return non_reduce_indices;
83  };

85  auto _compute_max = [&](const Array<Var>& indices) {
86  auto eval_range = insert_reduce_index(indices, k1);
87  return topi::MaxOp(x(eval_range), {k1});
88  };

90  auto _compute_exp = [&](const Tensor& max_elem, const Array<Var>& indices) {
91  auto non_reduce_indices = get_non_reduce_indices(indices);
92  return tvm::exp(x(indices) - max_elem(non_reduce_indices));
93  };

95  auto _compute_expsum = [&](const Tensor& exp, const Array<Var>& indices) {
96  auto eval_range = insert_reduce_index(indices, k2);
97  return tvm::sum(exp(eval_range), {k2});
98  };

100  auto _normalize = [&](const Tensor& exp, const Tensor& expsum, const Array<Var>& indices) {
101  auto non_reduce_indices = get_non_reduce_indices(indices);
102  return exp(indices) / expsum(non_reduce_indices);
103  };

124 inline Tensor log_softmax(const Tensor& x, std::string name = "tensor",
125  std::string tag = "log_softmax_output") {
126  CHECK_EQ(x->shape.size(), 2) << "Log softmax requires 2-D input";
127 
128  PrimExpr m = x->shape[0];
129  PrimExpr n = x->shape[1];
130 
131  auto k = tvm::te::reduce_axis(Range(0, n), "k");
132  auto max_elem =
133  tvm::te::compute({m}, [&](Var i) { return tvm::max(x(i, k), Array<IterVar>{k}); });
134  k = tvm::te::reduce_axis(Range(0, n), "k");
135 
136  auto expsum =
137  tvm::te::compute({m}, [&](Var i) { return tvm::sum(tvm::exp(x(i, k) - max_elem(i)), {k}); });
138 
139  return tvm::te::compute(
140  x->shape, [&](Var i, Var j) { return x(i, j) - max_elem(i) - tvm::log(expsum(i)); }, name,
141  tag);
