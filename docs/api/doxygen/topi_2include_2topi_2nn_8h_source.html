163 inline tvm::te::Tensor pad(const tvm::te::Tensor& t, const tvm::Array<tvm::PrimExpr>& pad_before,
164  tvm::Array<tvm::PrimExpr> pad_after = tvm::Array<tvm::PrimExpr>(),
165  PrimExpr pad_value = PrimExpr(), std::string name = "T_pad",
166  std::string tag = kElementWise, std::string pad_mode = "constant") {
167  if (pad_after.size() < pad_before.size()) {
168  for (size_t i = pad_after.size(); i < pad_before.size(); ++i) {
169  pad_after.push_back(pad_before[i]);
170  }
171  }
172  arith::Analyzer analyzer;
173  CHECK_GE(pad_before.size(), 1);
174  CHECK_EQ(pad_before.size(), pad_after.size());
175  tvm::Array<tvm::PrimExpr> output_shape;
176  tvm::Array<tvm::PrimExpr> pad_before_int32;
177  tvm::Array<tvm::PrimExpr> pad_after_int32;
178  for (const auto& ele : pad_before) {
179  pad_before_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
180  }
181  for (const auto& ele : pad_after) {
182  pad_after_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
183  }
184  for (size_t i = 0; i < t->shape.size(); ++i) {
185  if (i >= pad_before.size()) {
186  output_shape.push_back(t->shape[i]);
187  } else {
188  output_shape.push_back(
189  analyzer.Simplify(t->shape[i] + pad_before_int32[i] + pad_after_int32[i]));
190  }
191  }
192 
193  if (!pad_value.defined()) {
194  pad_value = tvm::tir::make_const(t->dtype, 0);
195  }
196 
197  auto l = [&](tvm::Array<tvm::tir::Var> ovars) {
198  tvm::Array<tvm::PrimExpr> indices;
199  tvm::Array<tvm::PrimExpr> sel;
200  tvm::Array<tvm::PrimExpr> pad_idx;
201  for (size_t i = 0; i < t->shape.size(); ++i) {
202  if (i >= pad_before_int32.size()) {
203  indices.push_back(ovars[i]);
204  continue;
205  }
206  if (!topi::detail::EqualCheck(pad_before_int32[i], 0)) {
207  sel.push_back(ovars[i] >= pad_before_int32[i]);
208  indices.push_back(ovars[i] - pad_before_int32[i]);
209  } else {
210  indices.push_back(ovars[i]);
211  }
212  if (!topi::detail::EqualCheck(pad_after_int32[i], 0)) {
213  sel.push_back(analyzer.Simplify(ovars[i] < pad_before_int32[i] + t->shape[i]));
214  }
215  if (pad_mode == "edge") {
216  pad_idx.push_back(
217  tvm::if_then_else(ovars[i] < pad_before[i], 0,
218  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
219  t->shape[i] - 1, ovars[i] - pad_before[i])));
220  } else if (pad_mode == "reflect") {
221  pad_idx.push_back(
222  tvm::if_then_else(ovars[i] < pad_before[i], pad_before[i] - ovars[i],
223  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
224  t->shape[i] * 2 - ovars[i] + pad_before[i] - 2,
225  ovars[i] - pad_before[i])));
226  }
227  }
228  if (sel.size() != 0) {
229  if (pad_mode == "constant") {
230  return tvm::if_then_else(detail::Map(sel, tvm::tir::AndNode::make), t(indices), pad_value);
231  } else if (pad_mode == "edge" || pad_mode == "reflect") {
232  return tvm::if_then_else(detail::Map(sel, tvm::tir::AndNode::make), t(indices), t(pad_idx));
233  }
234  }
235  return t(indices);
236  };
237  return tvm::te::compute(output_shape, l, name, tag);
238 }

260 inline tvm::te::Tensor conv2d_nchw(const tvm::te::Tensor& I, const tvm::te::Tensor& W,
261  int pad_h = 0, int pad_w = 0, int stride_h = 1, int stride_w = 1,
262  std::string name = "T_conv2d_nchw",
263  std::string tag = kConv2dNCHW) {
264  CHECK_EQ(4, I->shape.size());
265  CHECK_EQ(4, W->shape.size());
266  auto pH = I->shape[2];
267  auto pW = I->shape[3];
268  tvm::Array<tvm::PrimExpr> output_shape{
269  I->shape[0], // B
270  W->shape[0], // O
271  indexdiv(I->shape[2] - W->shape[2] + 2 * pad_h, stride_h) + 1, // H
272  indexdiv(I->shape[3] - W->shape[3] + 2 * pad_w, stride_w) + 1 // W
273  };
274  auto i = tvm::te::reduce_axis(tvm::Range{0, I->shape[1]}, "i");
275  auto kh = tvm::te::reduce_axis(tvm::Range{0, W->shape[2]}, "kh");
276  auto kw = tvm::te::reduce_axis(tvm::Range{0, W->shape[3]}, "kw");
277  auto T =
278  (pad_h == 0 && pad_w == 0) ? I : pad(I, {tvm::PrimExpr(0), tvm::PrimExpr(0), pad_h, pad_w});
279  auto l = [&](tvm::tir::Var b, tvm::tir::Var o, tvm::tir::Var h, tvm::tir::Var w) {
280  return tvm::sum(T(b, i, stride_h * h + kh, stride_w * w + kw) * W(o, i, kh, kw), {i, kh, kw});
281  };
282  return tvm::te::compute(output_shape, l, name, tag);
283 }
