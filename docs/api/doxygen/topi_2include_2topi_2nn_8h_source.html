150 inline tvm::te::Tensor pad(const tvm::te::Tensor& t, const tvm::Array<tvm::PrimExpr>& pad_before,
151  tvm::Array<tvm::PrimExpr> pad_after = tvm::Array<tvm::PrimExpr>(),
152  PrimExpr pad_value = PrimExpr(), std::string name = "T_pad",
153  std::string tag = kElementWise, std::string pad_mode = "constant") {
154  if (pad_after.size() < pad_before.size()) {
155  for (size_t i = pad_after.size(); i < pad_before.size(); ++i) {
156  pad_after.push_back(pad_before[i]);
157  }
158  }
159  arith::Analyzer analyzer;
160  CHECK_GE(pad_before.size(), 1);
161  CHECK_EQ(pad_before.size(), pad_after.size());
162  tvm::Array<tvm::PrimExpr> output_shape;
163  tvm::Array<tvm::PrimExpr> pad_before_int32;
164  tvm::Array<tvm::PrimExpr> pad_after_int32;
165  for (const auto& ele : pad_before) {
166  pad_before_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
167  }
168  for (const auto& ele : pad_after) {
169  pad_after_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
170  }
171  for (size_t i = 0; i < t->shape.size(); ++i) {
172  if (i >= pad_before.size()) {
173  output_shape.push_back(t->shape[i]);
174  } else {
175  output_shape.push_back(
176  analyzer.Simplify(t->shape[i] + pad_before_int32[i] + pad_after_int32[i]));
177  }
178  }
179 
180  if (!pad_value.defined()) {
181  pad_value = tvm::tir::make_const(t->dtype, 0);
182  }
183 
184  auto l = [&](tvm::Array<tvm::tir::Var> ovars) {
185  tvm::Array<tvm::PrimExpr> indices;
186  tvm::Array<tvm::PrimExpr> sel;
187  tvm::Array<tvm::PrimExpr> pad_idx;
188  for (size_t i = 0; i < t->shape.size(); ++i) {
189  if (i >= pad_before_int32.size()) {
190  indices.push_back(ovars[i]);
191  continue;
192  }
193  if (!topi::detail::EqualCheck(pad_before_int32[i], 0)) {
194  sel.push_back(ovars[i] >= pad_before_int32[i]);
195  indices.push_back(ovars[i] - pad_before_int32[i]);
196  } else {
197  indices.push_back(ovars[i]);
198  }
199  if (!topi::detail::EqualCheck(pad_after_int32[i], 0)) {
200  sel.push_back(analyzer.Simplify(ovars[i] < pad_before_int32[i] + t->shape[i]));
201  }
202  if (pad_mode == "edge") {
203  pad_idx.push_back(
204  tvm::if_then_else(ovars[i] < pad_before[i], 0,
205  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
206  t->shape[i] - 1, ovars[i] - pad_before[i])));
207  } else if (pad_mode == "reflect") {
208  pad_idx.push_back(
209  tvm::if_then_else(ovars[i] < pad_before[i], pad_before[i] - ovars[i],
210  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
211  t->shape[i] * 2 - ovars[i] + pad_before[i] - 2,
212  ovars[i] - pad_before[i])));
213  }
214  }
215  if (sel.size() != 0) {
216  auto fand = [](PrimExpr a, PrimExpr b) { return a && b; };
217  if (pad_mode == "constant") {
218  return tvm::if_then_else(foldl(fand, const_true(1), sel), t(indices), pad_value);
219  } else if (pad_mode == "edge" || pad_mode == "reflect") {
220  return tvm::if_then_else(foldl(fand, const_true(1), sel), t(indices), t(pad_idx));
221  }
222  }
223  return t(indices);
224  };
225  return tvm::te::compute(output_shape, l, name, tag);
226 }

248 inline tvm::te::Tensor conv2d_nchw(const tvm::te::Tensor& I, const tvm::te::Tensor& W,
249  int pad_h = 0, int pad_w = 0, int stride_h = 1, int stride_w = 1,
250  std::string name = "T_conv2d_nchw",
251  std::string tag = kConv2dNCHW) {
252  CHECK_EQ(4, I->shape.size());
253  CHECK_EQ(4, W->shape.size());
254  auto pH = I->shape[2];
255  auto pW = I->shape[3];
256  tvm::Array<tvm::PrimExpr> output_shape{
257  I->shape[0], // B
258  W->shape[0], // O
259  indexdiv(I->shape[2] - W->shape[2] + 2 * pad_h, stride_h) + 1, // H
260  indexdiv(I->shape[3] - W->shape[3] + 2 * pad_w, stride_w) + 1 // W
261  };
262  auto i = tvm::te::reduce_axis(tvm::Range{0, I->shape[1]}, "i");
263  auto kh = tvm::te::reduce_axis(tvm::Range{0, W->shape[2]}, "kh");
264  auto kw = tvm::te::reduce_axis(tvm::Range{0, W->shape[3]}, "kw");
265  auto T =
266  (pad_h == 0 && pad_w == 0) ? I : pad(I, {tvm::PrimExpr(0), tvm::PrimExpr(0), pad_h, pad_w});
267  auto l = [&](tvm::tir::Var b, tvm::tir::Var o, tvm::tir::Var h, tvm::tir::Var w) {
268  return tvm::sum(T(b, i, stride_h * h + kh, stride_w * w + kw) * W(o, i, kh, kw), {i, kh, kw});
269  };
270  return tvm::te::compute(output_shape, l, name, tag);
271 }
