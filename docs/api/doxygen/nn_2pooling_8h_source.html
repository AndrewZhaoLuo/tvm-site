63 inline Tensor pool_impl(const Tensor& x, const Array<PrimExpr>& kernel_size,
64  const Array<PrimExpr>& stride_size, const Array<PrimExpr>& padding_size,
65  PoolType pool_type, bool ceil_mode, const size_t height_axis,
66  const size_t width_axis, bool count_include_pad) {
67  CHECK(x->shape.size() >= 2) << "Pooling input must >= 2-D (H, W)";
68  CHECK_EQ(kernel_size.size(), 2) << "Pooling kernel_size must have 2 elements";
69  CHECK_EQ(stride_size.size(), 2) << "Pooling stride_size must have 2 elements";
70  CHECK_EQ(padding_size.size(), 4) << "Pooling padding_size must have 4 elements";
71 
72  auto kernel_height = cast(DataType::DataType::Int(32), kernel_size[0]);
73  auto kernel_width = cast(DataType::DataType::Int(32), kernel_size[1]);
74  auto stride_height = cast(DataType::DataType::Int(32), stride_size[0]);
75  auto stride_width = cast(DataType::DataType::Int(32), stride_size[1]);
76 
77  auto height = x->shape[height_axis];
78  auto width = x->shape[width_axis];
79 
80  auto pad_top = cast(DataType::DataType::Int(32), padding_size[0]);
81  auto pad_left = cast(DataType::DataType::Int(32), padding_size[1]);
82  auto pad_bottom = cast(DataType::DataType::Int(32), padding_size[2]);
83  auto pad_right = cast(DataType::DataType::Int(32), padding_size[3]);
84 
85  if (ceil_mode) {
86  // Additional padding to ensure we do ceil instead of floor when
87  // dividing by stride.
88  pad_bottom += stride_height - 1;
89  pad_right += stride_width - 1;
90  }
91 
92  Array<PrimExpr> pad_before(std::vector<PrimExpr>(x->shape.size(), 0));
93  pad_before.Set(height_axis, pad_top);
94  pad_before.Set(width_axis, pad_left);
95 
96  Array<PrimExpr> pad_after(std::vector<PrimExpr>(x->shape.size(), 0));
97  pad_after.Set(height_axis, pad_bottom);
98  pad_after.Set(width_axis, pad_right);
99  arith::Analyzer analyzer;
100  auto out_height =
101  analyzer.Simplify(indexdiv(height - kernel_height + pad_top + pad_bottom, stride_height) + 1);
102  auto out_width =
103  analyzer.Simplify(indexdiv(width - kernel_width + pad_left + pad_right, stride_width) + 1);
104 
105  auto dheight = tvm::te::reduce_axis(Range(0, kernel_height));
106  auto dwidth = tvm::te::reduce_axis(Range(0, kernel_width));
107 
108  Array<PrimExpr> out_shape = x->shape;
109  out_shape.Set(height_axis, out_height);
110  out_shape.Set(width_axis, out_width);
111 
112  const int64_t* padding_h0 = as_const_int(pad_top);
113  const int64_t* padding_w0 = as_const_int(pad_left);
114  const int64_t* padding_h1 = as_const_int(pad_bottom);
115  const int64_t* padding_w1 = as_const_int(pad_right);
116  const bool do_pad = ((padding_h0 && *padding_h0) || (padding_w0 && *padding_w0)) ||
117  ((padding_h1 && *padding_h1) || (padding_w1 && *padding_w1));
118 
119  if (pool_type == kMaxPool) {
120  auto temp = do_pad ? pad(x, pad_before, pad_after, tvm::min_value(x->dtype), "pad_temp") : x;
121  return tvm::te::compute(
122  out_shape,
123  [&](const Array<Var>& output) {
124  Array<PrimExpr> indices;
125  for (const Var& var : output) indices.push_back(var);
126  indices.Set(height_axis, output[height_axis] * stride_height + dheight);
127  indices.Set(width_axis, output[width_axis] * stride_width + dwidth);
128  return tvm::max(temp(indices), {dheight, dwidth});
129  },
130  "tensor", "pool_max");
131  } else if (pool_type == kAvgPool) {
132  // Pad the inputs
133  auto temp = do_pad ? pad(x, pad_before, pad_after, 0, "pad_temp") : x;
134 
135  // TVM compute for summing the pooling window.
136  auto pool_sum = tvm::te::compute(
137  out_shape,
138  [&](const Array<Var>& output) {
139  Array<PrimExpr> indices;
140  for (const Var& var : output) indices.push_back(var);
141  indices.Set(height_axis, output[height_axis] * stride_height + dheight);
142  indices.Set(width_axis, output[width_axis] * stride_width + dwidth);
143  return tvm::sum(temp(indices), {dheight, dwidth});
144  },
145  "tensor", "pool_sum");
146 
147  // TVM compute for dividing the reduced window sum by kernel size.
148  return tvm::te::compute(
149  out_shape,
150  [&](const Array<Var>& output) {
151  Array<PrimExpr> indices;
152  for (const Var& var : output) indices.push_back(var);
153  if (count_include_pad) {
154  return div(pool_sum(indices), (kernel_height * kernel_width));
155  } else {
156  PrimExpr h_start = output[height_axis] * stride_height - pad_top;
157  PrimExpr w_start = output[width_axis] * stride_width - pad_left;
158 
159  PrimExpr h_end = min(h_start + kernel_height, height);
160  PrimExpr w_end = min(w_start + kernel_width, width);
161  h_start = max(h_start, make_const(DataType::DataType::Int(32), 0));
162  w_start = max(w_start, make_const(DataType::DataType::Int(32), 0));
163  PrimExpr divide_factor = max((h_end - h_start) * (w_end - w_start),
164  make_const(DataType::DataType::Int(32), 1));
165  return div(pool_sum(indices), divide_factor);
166  }
167  },
168  "tensor", kElementWise);
169  } else {
170  LOG(ERROR) << "Unrecognized pool_type: " << pool_type;
171  return x;
172  }

175 inline Tensor pool_grad_impl(const Tensor& out_grad, const Tensor& x,
176  const Array<PrimExpr>& kernel_size, const Array<PrimExpr>& stride_size,
177  const Array<PrimExpr>& padding_size, PoolType pool_type,
178  bool ceil_mode, const size_t height_axis, const size_t width_axis,
179  bool count_include_pad) {
180  CHECK(out_grad->shape.size() >= 2) << "Pooling grad output must >= 2-D (H, W)";
181  CHECK(x->shape.size() >= 2) << "Pooling input must >= 2-D (H, W)";
182  CHECK_EQ(kernel_size.size(), 2) << "Pooling kernel_size must have 2 elements";
183  CHECK_EQ(stride_size.size(), 2) << "Pooling stride_size must have 2 elements";
184  CHECK_EQ(padding_size.size(), 4) << "Pooling padding_size must have 4 elements";
185 
186  auto kernel_height = cast(DataType::DataType::Int(32), kernel_size[0]);
187  auto kernel_width = cast(DataType::DataType::Int(32), kernel_size[1]);
188  auto stride_height = cast(DataType::DataType::Int(32), stride_size[0]);
189  auto stride_width = cast(DataType::DataType::Int(32), stride_size[1]);
190 
191  auto height = x->shape[height_axis];
192  auto width = x->shape[width_axis];
193 
194  auto pad_top = cast(DataType::DataType::Int(32), padding_size[0]);
195  auto pad_left = cast(DataType::DataType::Int(32), padding_size[1]);
196  auto pad_bottom = cast(DataType::DataType::Int(32), padding_size[2]);
197  auto pad_right = cast(DataType::DataType::Int(32), padding_size[3]);
198 
199  if (ceil_mode) {
200  // Additional padding to ensure we do ceil instead of floor when
201  // dividing by stride.
202  pad_bottom += stride_height - 1;
203  pad_right += stride_width - 1;
204  }
205 
206  Array<PrimExpr> pad_before(std::vector<PrimExpr>(x->shape.size(), 0));
207  pad_before.Set(height_axis, pad_top);
208  pad_before.Set(width_axis, pad_left);
209 
210  Array<PrimExpr> pad_after(std::vector<PrimExpr>(x->shape.size(), 0));
211  pad_after.Set(height_axis, pad_bottom);
212  pad_after.Set(width_axis, pad_right);
213  arith::Analyzer analyzer;
214  auto out_height =
215  analyzer.Simplify((height - kernel_height + pad_top + pad_bottom) / stride_height + 1);
216  auto out_width =
217  analyzer.Simplify((width - kernel_width + pad_left + pad_right) / stride_width + 1);
218 
219  auto dheight = tvm::te::reduce_axis(Range(0, kernel_height));
220  auto dwidth = tvm::te::reduce_axis(Range(0, kernel_width));
221 
222  Array<PrimExpr> out_shape = x->shape;
223  out_shape.Set(height_axis, out_height);
224  out_shape.Set(width_axis, out_width);
225 
226  const int64_t* padding_h0 = as_const_int(pad_top);
227  const int64_t* padding_w0 = as_const_int(pad_left);
228  const int64_t* padding_h1 = as_const_int(pad_bottom);
229  const int64_t* padding_w1 = as_const_int(pad_right);
230  const bool do_pad = ((padding_h0 && *padding_h0) || (padding_w0 && *padding_w0)) ||
231  ((padding_h1 && *padding_h1) || (padding_w1 && *padding_w1));
232 
233  if (pool_type == kMaxPool) {
234  Array<PrimExpr> ravel_shape{x->shape.begin(), x->shape.end()};
235  ravel_shape.Set(height_axis, ravel_shape[height_axis] + pad_top + pad_bottom);
236  ravel_shape.Set(width_axis, ravel_shape[width_axis] + pad_left + pad_right);
237 
238  auto windowh =
239  tvm::te::reduce_axis(Range(0, (kernel_height + stride_height - 1) / stride_height));
240  auto windoww = tvm::te::reduce_axis(Range(0, (kernel_width + stride_width - 1) / stride_width));
241 
242  auto argmax = MakeArgmaxReducer();
243  auto pad_x = do_pad ? pad(x, pad_before, pad_after, tvm::min_value(x->dtype), "pad_temp") : x;
244 
245  auto mp_argmax = tvm::te::compute(
246  out_shape,
247  [&](const Array<Var>& inds) {
248  Array<PrimExpr> window_inds{inds.begin(), inds.end()};
249  window_inds.Set(height_axis, inds[height_axis] * stride_height + dheight);
250  window_inds.Set(width_axis, inds[width_axis] * stride_width + dwidth);
251  auto idx = detail::RavelIndex(window_inds, ravel_shape);
252  return argmax({idx, pad_x(window_inds)}, {dheight, dwidth}, nullptr);
253  },
254  "maxpool_grad_argmax", kCommReduceIdx);
255 
256  auto mp_inds = mp_argmax[0];
257 
258  return tvm::te::compute(
259  x->shape,
260  [&](const Array<Var>& inds) {
261  Array<PrimExpr> pad_inds{inds.begin(), inds.end()};
262  pad_inds.Set(height_axis, pad_inds[height_axis] + pad_top);
263  pad_inds.Set(width_axis, pad_inds[width_axis] + pad_left);
264  auto idx = detail::RavelIndex(pad_inds, ravel_shape);
265 
266  Array<PrimExpr> out_idx{inds.begin(), inds.end()};
267  out_idx.Set(height_axis, (inds[height_axis] + pad_top) / stride_height - windowh);
268  out_idx.Set(width_axis, (inds[width_axis] + pad_left) / stride_width - windoww);
269 
270  PrimExpr out_idx_lower_h = tir::Select(
271  pad_inds[height_axis] < kernel_height, make_const(DataType::DataType::Int(32), 0),
272  (pad_inds[height_axis] - kernel_height) / stride_height + 1);
273  PrimExpr out_idx_lower_w = tir::Select(
274  pad_inds[width_axis] < kernel_width, make_const(DataType::DataType::Int(32), 0),
275  (pad_inds[width_axis] - kernel_width) / stride_width + 1);
276 
277  return tvm::sum(
278  tvm::if_then_else(tir::And(tir::And(out_idx[height_axis] >= out_idx_lower_h,
279  out_idx[width_axis] >= out_idx_lower_w),
280  mp_inds(out_idx) == idx),
281  out_grad(out_idx), make_const(x->dtype, 0)),
282  {windowh, windoww});
283  },
284  "T_pool_grad", "pool_grad_max");
285  } else if (pool_type == kAvgPool) {
286  auto windowh =
287  tvm::te::reduce_axis(Range(0, (kernel_height + stride_height - 1) / stride_height));
288  auto windoww = tvm::te::reduce_axis(Range(0, (kernel_width + stride_width - 1) / stride_width));
289  return tvm::te::compute(
290  x->shape,
291  [&](const Array<Var>& inds) {
292  PrimExpr pad_h_idx = inds[height_axis] + pad_top;
293  PrimExpr pad_w_idx = inds[width_axis] + pad_left;
294 
295  // output indices whose pooling windows cover current input element (can be out-of-bound)
296  Array<PrimExpr> out_idx{inds.begin(), inds.end()};
297  out_idx.Set(height_axis, (pad_h_idx / stride_height - windowh));
298  out_idx.Set(width_axis, (pad_w_idx / stride_width - windoww));
299 
300  PrimExpr out_idx_lower_h =
301  tir::Select(pad_h_idx < kernel_height, make_const(DataType::Int(32), 0),
302  (pad_h_idx - kernel_height) / stride_height + 1);
303  PrimExpr out_idx_lower_w =
304  tir::Select(pad_w_idx < kernel_width, make_const(DataType::Int(32), 0),
305  (pad_w_idx - kernel_width) / stride_width + 1);
306 
307  PrimExpr divide_factor; // number of pooled elements
308  if (count_include_pad) {
309  divide_factor = kernel_height * kernel_width;
310  } else {
311  PrimExpr h_start = out_idx[height_axis] * stride_height - pad_top;
312  PrimExpr w_start = out_idx[width_axis] * stride_width - pad_left;
313 
314  PrimExpr h_end = min(h_start + kernel_height, height);
315  PrimExpr w_end = min(w_start + kernel_width, width);
316  h_start = max(h_start, make_const(DataType::Int(32), 0));
317  w_start = max(w_start, make_const(DataType::Int(32), 0));
318  divide_factor =
319  max((h_end - h_start) * (w_end - w_start), make_const(DataType::Int(32), 1));
320  }
321  return tvm::sum(
322  tvm::if_then_else(tir::And(tir::And(out_idx[height_axis] >= out_idx_lower_h,
323  out_idx[height_axis] < out_height),
324  tir::And(out_idx[width_axis] >= out_idx_lower_w,
325  out_idx[width_axis] < out_width)),
326  out_grad(out_idx) / divide_factor, make_const(out_grad->dtype, 0)),
327  {windowh, windoww});
328  },
329  "T_pool_grad", "pool_grad_avg");
330  } else {
331  LOG(ERROR) << "Unrecognized pool_type: " << pool_type;
332  return Tensor();
333  }

336 inline bool find_depth_height_width(const std::string& layout, int* depth_axis, int* height_axis,
337  int* width_axis) {
338  *depth_axis = -1, *height_axis = -1, *width_axis = -1;
339  int curr_idx = 0;
340  for (size_t i = 0; i < layout.size(); ++i) {
341  if ((layout[i] >= 'A' && layout[i] <= 'Z') || (layout[i] >= 'a' && layout[i] <= 'z')) {
342  if (layout[i] == 'D') {
343  if (*depth_axis != -1) return false;
344  *depth_axis = curr_idx;
345  } else if (layout[i] == 'H') {
346  if (*height_axis != -1) return false;
347  *height_axis = curr_idx;
348  } else if (layout[i] == 'W') {
349  if (*width_axis != -1) return false;
350  *width_axis = curr_idx;
351  } else if (layout[i] == 'd' || layout[i] == 'h' || layout[i] == 'w') {
352  // do not support split on height or width, e.g., NCHW16w
353  return false;
354  }
355  ++curr_idx;
356  }
357  }
358  if (*depth_axis == -1 || *height_axis == -1 || *width_axis == -1) return false;
359  return true;
360 }

478 inline Tensor adaptive_pool_impl(const Tensor& x, const Array<PrimExpr>& output_size,
479  PoolType pool_type, const std::vector<int>& axes) {
480  const auto n_dim = output_size.size();
481  CHECK_EQ(axes.size(), n_dim) << "The number of axes not equal to the in/out dimension";
482 
483  Array<PrimExpr> out_shape = x->shape;
484  Array<PrimExpr> in_size, out_size;
485  for (size_t i = 0; i < n_dim; ++i) {
486  in_size.push_back(x->shape[axes[i]]);
487  out_size.push_back(cast(DataType::Int(32), output_size[i]));
488  out_shape.Set(axes[i], out_size[i]);
489  }
490 
491  auto get_iter_vars = [=](const Array<Var>& output, bool reduce_indices) {
492  Array<PrimExpr> indices;
493  for (size_t i = 0; i < output.size(); ++i) indices.push_back(output[i]);
494  Array<tir::IterVar> reduce_axes;
495  for (size_t i = 0; i < n_dim; ++i) {
496  auto i_start = start_index(output[axes[i]], out_size[i], in_size[i]);
497  auto i_end = end_index(output[axes[i]], out_size[i], in_size[i]);
498  auto rv_name = "rv" + std::to_string(i);
499  auto rv_axis = tvm::te::reduce_axis(Range(0, i_end - i_start), rv_name);
500  reduce_axes.push_back(rv_axis);
501  if (reduce_indices) {
502  indices.Set(axes[i], i_start + rv_axis);
503  }
504  }
505  return std::make_tuple(indices, reduce_axes);
506  };
507 
508  if (pool_type == kMaxPool) {
509  return tvm::te::compute(
510  out_shape,
511  [&](const Array<Var>& output) {
512  Array<PrimExpr> indices;
513  Array<tir::IterVar> reduce_axes;
514  std::tie(indices, reduce_axes) = get_iter_vars(output, true);
515  return tvm::max(x(indices), reduce_axes); // NOLINT(*)
516  },
517  "tensor", "adaptive_pool_max");
518  } else if (pool_type == kAvgPool) {
519  auto pool_sum = tvm::te::compute(
520  out_shape,
521  [&](const Array<Var>& output) {
522  Array<PrimExpr> indices;
523  Array<tir::IterVar> reduce_axes;
524  std::tie(indices, reduce_axes) = get_iter_vars(output, true);
525  return tvm::sum(x(indices), reduce_axes);
526  },
527  "tensor", "adaptive_pool_sum");
528 
529  return tvm::te::compute(
530  out_shape,
531  [&](const Array<Var>& output) {
532  Array<PrimExpr> indices;
533  Array<tir::IterVar> reduce_axes;
534  std::tie(indices, reduce_axes) = get_iter_vars(output, false);
535 
536  PrimExpr divide_factor = tvm::cast(x->dtype, 1);
537  for (size_t i = 0; i < n_dim; ++i) {
538  divide_factor *= tvm::cast(x->dtype, reduce_axes[i]->dom->extent);
539  }
540 
541  return div(pool_sum(indices), divide_factor);
542  },
543  "tensor", kElementWise);
544  } else {
545  LOG(ERROR) << "Unrecognized pool_type: " << pool_type;
546  return x;
547  }
548 }

644  const Array<PrimExpr>& stride_size, const Array<PrimExpr>& padding_size,
645  PoolType pool_type, bool ceil_mode, const std::vector<int>& axis,
646  bool count_include_pad) {
647  int k_size = kernel_size.size();
648  int x_size = x->shape.size();
649  CHECK_EQ(stride_size.size(), k_size) << "Pooling stride_size must have same elements as kernel";
650  CHECK_EQ(padding_size.size(), k_size * 2) << "Pooling padding_size must has double elements of"
651  " kernel";
652  CHECK_EQ(axis.size(), k_size) << "axis must have same elements as kernel";
653 
654  Array<IterVar> daxis;
655  std::vector<PrimExpr> kernel(k_size);
656  std::vector<PrimExpr> stride(k_size);
657  std::vector<PrimExpr> pad_head(k_size);
658  std::vector<PrimExpr> pad_tail(k_size);
659  Array<PrimExpr> pad_before(std::vector<PrimExpr>(x_size, 0));
660  Array<PrimExpr> pad_after(std::vector<PrimExpr>(x_size, 0));
661  Array<PrimExpr> out_shape = x->shape;
662 
663  bool do_pad = false;
664  for (int i = 0; i < k_size; i++) {
665  int ii = axis[i];
666  kernel[i] = cast(DataType::Int(32), kernel_size[i]);
667  stride[i] = cast(DataType::Int(32), stride_size[i]);
668  pad_head[i] = cast(DataType::Int(32), padding_size[i]);
669  pad_tail[i] = cast(DataType::Int(32), padding_size[i + k_size]);
670  const int64_t* padding0 = as_const_int(pad_head[i]);
671  const int64_t* padding1 = as_const_int(pad_tail[i]);
672  do_pad = (do_pad) ? do_pad : ((padding0 && *padding0) || (padding1 && *padding1));
673 
674  if (ceil_mode) {
675  // Additional padding to ensure we do ceil instead of floor when
676  // dividing by stride.
677  pad_tail[i] += stride[i] - 1;
678  }
679 
680  daxis.push_back(tvm::te::reduce_axis(Range(0, kernel[i])));
681 
682  pad_before.Set(ii, pad_head[i]);
683  pad_after.Set(ii, pad_tail[i]);
684 
685  arith::Analyzer analyzer;
686  auto out_dim = analyzer.Simplify(
687  indexdiv(x->shape[ii] - kernel[i] + pad_head[i] + pad_tail[i], stride[i]) + 1);
688 
689  out_shape.Set(ii, out_dim);
690  }
691 
692  if (pool_type == kMaxPool) {
693  auto temp = do_pad ? pad(x, pad_before, pad_after, tvm::min_value(x->dtype), "pad_temp") : x;
694  return tvm::te::compute(
695  out_shape,
696  [&](const Array<Var>& output) {
697  Array<PrimExpr> indices;
698  for (const Var& var : output) indices.push_back(var);
699 
700  for (int i = 0; i < k_size; i++) {
701  int ii = axis[i];
702  indices.Set(ii, output[ii] * stride[i] + daxis[i]);
703  }
704 
705  return tvm::max(temp(indices), daxis);
706  },
707  "tensor", "pool_max");
708  } else if (pool_type == kAvgPool) {
709  // Pad the inputs
710  auto temp = do_pad ? pad(x, pad_before, pad_after, 0, "pad_temp") : x;
711 
712  // TVM compute for summing the pooling window.
713  auto pool_sum = tvm::te::compute(
714  out_shape,
715  [&](const Array<Var>& output) {
716  Array<PrimExpr> indices;
717  for (const Var& var : output) indices.push_back(var);
718 
719  for (int i = 0; i < k_size; i++) {
720  int ii = axis[i];
721  indices.Set(ii, output[ii] * stride[i] + daxis[i]);
722  }
723  return tvm::sum(temp(indices), daxis);
724  },
725  "tensor", "pool_sum");
726 
727  // TVM compute for dividing the reduced window sum by kernel size.
728  return tvm::te::compute(
729  out_shape,
730  [&](const Array<Var>& output) {
731  Array<PrimExpr> indices;
732  for (const Var& var : output) indices.push_back(var);
733  if (count_include_pad) {
734  auto kernel_size = make_const(DataType::Int(32), 1);
735  for (int i = 0; i < k_size; i++) {
736  kernel_size *= kernel[i];
737  }
738  return div(pool_sum(indices), kernel_size);
739  } else {
740  std::vector<PrimExpr> start(k_size);
741  std::vector<PrimExpr> end(k_size);
742  auto kernel_size = make_const(DataType::Int(32), 1);
743  for (int i = 0; i < k_size; i++) {
744  int ii = axis[i];
745  start[i] = output[ii] * stride[i] - pad_head[i];
746  end[i] = min(start[i] + kernel[i], x->shape[ii]);
747  start[i] = max(start[i], make_const(DataType::Int(32), 0));
748  kernel_size *= (end[i] - start[i]);
749  }
750 
751  PrimExpr divide_factor = max(kernel_size, make_const(DataType::Int(32), 1));
752  return div(pool_sum(indices), divide_factor);
753  }
754  },
755  "tensor", kElementWise);
756  } else {
757  LOG(ERROR) << "Unrecognized pool_type: " << pool_type;
758  return x;
759  }
760 }
