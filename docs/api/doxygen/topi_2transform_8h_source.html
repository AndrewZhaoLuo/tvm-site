601 inline Tensor strided_slice(const Tensor& x, const Array<Integer>& begin, const Array<Integer>& end,
602  const Array<Integer>& strides, std::string slice_mode = "end",
603  std::string name = "T_strided_slice", std::string tag = kInjective) {

605  // Setup the ranges.
606  // NOTE: this code duplicates the shape inference logic relay.op
607  // Consider to refactor in the future.
608  std::vector<int64_t> stride_vec(src_tensor_dim, 1);
609  for (size_t i = 0; i < strides.size(); ++i) {
610  ICHECK(strides[i].defined());
611  stride_vec[i] = strides[i]->value;
612  }
613 
614  const int64_t max_range = std::numeric_limits<int64_t>::max();
615 
616  std::vector<int64_t> begin_vec;
617  for (size_t i = 0; i < begin.size(); ++i) {
618  if (!begin[i].defined()) {
619  // value=None
620  begin_vec.push_back(stride_vec[i] > 0 ? 0 : max_range);
621  } else {
622  begin_vec.push_back(begin[i]->value);
623  }
624  }
625  for (size_t i = begin_vec.size(); i < src_tensor_dim; ++i) {
626  begin_vec.push_back(stride_vec[i] > 0 ? 0 : max_range);
627  }
628 
629  std::vector<int64_t> end_vec;
630  for (size_t i = 0; i < end.size(); ++i) {
631  // allow end to be None
632 
633  if (!end[i].defined()) {
634  end_vec.push_back(stride_vec[i] < 0 ? 0 : max_range);
635  } else if (slice_mode == "size") {
636  if (end[i]->value < 0) {
637  end_vec.push_back(stride_vec[i] < 0 ? 0 : max_range);
638  } else {
639  end_vec.push_back(begin_vec[i] + end[i]->value);
640  }
641  } else {
642  end_vec.push_back(end[i]->value);
643  }
644  }
645  for (size_t i = end_vec.size(); i < src_tensor_dim; ++i) {
646  end_vec.push_back(stride_vec[i] < 0 ? 0 : max_range);
647  }
648  // Compute
649  Array<PrimExpr> out_shape;
650  Array<PrimExpr> begin_expr;
651  Array<PrimExpr> strides_expr;
652 
653  for (size_t i = 0; i < src_tensor_dim; ++i) {
654  int64_t begin_range = stride_vec[i] < 0 ? -1 : 0;
655  int64_t dim_i = GetConstInt(x->shape[i]);
656  int64_t end_range = stride_vec[i] < 0 ? dim_i - 1 : dim_i;
657  // transform negative indices to positive value, clips on the correct range

659  if (index < 0) {
660  index += dim_i;
661  }
662  return std::min(std::max(index, begin_range), end_range);
663  };
664

666  int64_t end_i = index_canonicalization(end_vec[i]);
667 
668  int interval = std::abs(end_i - begin_i);
669  int slice_size =
670  static_cast<int>((interval + std::abs(stride_vec[i]) - 1) / std::abs(stride_vec[i]));
671  ICHECK(stride_vec[i] < 0 ? (end_i <= begin_i) : (begin_i <= end_i))
672  << ": Input [Begin=" << begin_vec[i] << ", End=" << end_vec[i]
673  << "] is invalid for axis=" << i;
674

676  strides_expr.push_back(

678  out_shape.push_back(slice_size);

684  Array<PrimExpr> real_indices;
685  for (size_t i = 0; i < src_tensor_dim; ++i) {
686  real_indices.push_back(indices[i] * strides_expr[i] + begin_expr[i]);
687  }
688  return x(real_indices);
689  },
690  name, tag);
