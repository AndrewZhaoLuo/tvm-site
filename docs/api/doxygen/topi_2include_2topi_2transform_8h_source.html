572

574  for (size_t i = 0; i < begin.size(); ++i) {

592  } else if (slice_mode == "size") {

594  end_vec.push_back(stride_vec[i] < 0 ? 0 : max_range);
595  } else {
596  end_vec.push_back(begin_vec[i] + end[i]->value);
597  }
598  } else {
599  end_vec.push_back(end[i]->value);
600  }
601  }

623  int64_t end_i = index_canonicalization(end_vec[i]);
