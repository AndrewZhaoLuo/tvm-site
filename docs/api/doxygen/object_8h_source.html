800 template <typename TargetType>
801 inline bool Object::IsInstance() const {
802  const Object* self = this;
803  // NOTE: the following code can be optimized by
804  // compiler dead-code elimination for already known constants.
805  if (self != nullptr) {
806  // Everything is a subclass of object.
807  if (std::is_same<TargetType, Object>::value) return true;
808  if (TargetType::_type_final) {
809  // if the target type is a final type
810  // then we only need to check the equivalence.
811  return self->type_index_ == TargetType::RuntimeTypeIndex();
812  } else {
813  // if target type is a non-leaf type
814  // Check if type index falls into the range of reserved slots.
815  uint32_t begin = TargetType::RuntimeTypeIndex();
816  // The condition will be optimized by constant-folding.
817  if (TargetType::_type_child_slots != 0) {
818  uint32_t end = begin + TargetType::_type_child_slots;
819  if (self->type_index_ >= begin && self->type_index_ < end) return true;
820  } else {
821  if (self->type_index_ == begin) return true;
822  }
823  if (!TargetType::_type_child_slots_can_overflow) return false;
824  // Invariance: parent index is always smaller than the child.
825  if (self->type_index_ < TargetType::RuntimeTypeIndex()) return false;
826  // The rare slower-path, check type hierachy.
827  return self->DerivedFrom(TargetType::RuntimeTypeIndex());
828  }
829  } else {
830  return false;
831  }
832 }
