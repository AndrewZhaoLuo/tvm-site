819 template <typename TargetType>
820 inline bool Object::IsInstance() const {
821  const Object* self = this;
822  // NOTE: the following code can be optimized by
823  // compiler dead-code elimination for already known constants.
824  if (self != nullptr) {
825  // Everything is a subclass of object.
826  if (std::is_same<TargetType, Object>::value) return true;
827  if (TargetType::_type_final) {
828  // if the target type is a final type
829  // then we only need to check the equivalence.
830  return self->type_index_ == TargetType::RuntimeTypeIndex();
831  } else {
832  // if target type is a non-leaf type
833  // Check if type index falls into the range of reserved slots.
834  uint32_t begin = TargetType::RuntimeTypeIndex();
835  // The condition will be optimized by constant-folding.
836  if (TargetType::_type_child_slots != 0) {
837  uint32_t end = begin + TargetType::_type_child_slots;
838  if (self->type_index_ >= begin && self->type_index_ < end) return true;
839  } else {
840  if (self->type_index_ == begin) return true;
841  }
842  if (!TargetType::_type_child_slots_can_overflow) return false;
843  // Invariance: parent index is always smaller than the child.
844  if (self->type_index_ < TargetType::RuntimeTypeIndex()) return false;
845  // The rare slower-path, check type hierarchy.
846  return self->DerivedFrom(TargetType::RuntimeTypeIndex());
847  }
848  } else {
849  return false;
850  }
851 }
