96  bool SEqualReduce(const BufferNode* other, SEqualReducer equal) const {
97  // Use DefEqual as buffer can define variables

99  return equal.DefEqual(data, other->data) && equal(dtype, other->dtype) &&
100  equal.DefEqual(shape, other->shape) && equal.DefEqual(strides, other->strides) &&
101  equal.DefEqual(elem_offset, other->elem_offset) && equal(scope, other->scope) &&
