332 inline DLDataType String2DLDataType(std::string s) {
333  DLDataType t;
334  // handle void type
335  if (s.length() == 0) {
336  t = DataType::Void();
337  return t;
338  }
339  t.bits = 32;
340  t.lanes = 1;
341  const char* scan;
342  if (s.substr(0, 3) == "int") {
343  t.code = kDLInt;
344  scan = s.c_str() + 3;
345  } else if (s.substr(0, 4) == "uint") {
346  t.code = kDLUInt;
347  scan = s.c_str() + 4;
348  } else if (s.substr(0, 5) == "float") {
349  t.code = kDLFloat;
350  scan = s.c_str() + 5;
351  } else if (s.substr(0, 6) == "handle") {
352  t.code = kTVMOpaqueHandle;
353  t.bits = 64; // handle uses 64 bit by default.
354  scan = s.c_str() + 6;
355  } else if (s == "bool") {
356  t.code = kDLUInt;
357  t.bits = 1;
358  t.lanes = 1;
359  return t;
360  } else if (s.substr(0, 6) == "bfloat") {
361  t.code = DataType::kBFloat;
362  scan = s.c_str() + 6;
363  } else if (s.substr(0, 6) == "custom") {
364  t.code = ParseCustomDatatype(s, &scan);
365  } else {
366  scan = s.c_str();
367  LOG(FATAL) << "unknown type " << s;
368  }
369  char* xdelim; // emulate sscanf("%ux%u", bits, lanes)
370  uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
371  if (bits != 0) t.bits = bits;
372  char* endpt = xdelim;
373  if (*xdelim == 'x') {
374  t.lanes = static_cast<uint16_t>(strtoul(xdelim + 1, &endpt, 10));
375  }
376  ICHECK(endpt == s.c_str() + s.length()) << "unknown type " << s;
377  return t;
378 }
