792 template <typename ValueType>
793 inline PrimExpr MakeConstScalar(DataType t, ValueType value) {
794  if (t.is_int()) return IntImm(t, static_cast<int64_t>(value));
795  if (t.is_uint()) {
796  // Use IntImm if it is a small integer
797  uint64_t uval = static_cast<uint64_t>(value);
798  if (uval <= static_cast<uint64_t>(std::numeric_limits<int64_t>::max())) {
799  return IntImm(t, static_cast<int64_t>(value));
800  } else {
801  uint64_t mask = (static_cast<uint64_t>(1) << 32U) - 1U;
802  uint64_t low = uval & mask;
803  uint64_t high = uval >> 32U;
804  return LargeUIntImm(t, static_cast<int64_t>(low), static_cast<int64_t>(high));
805  }
806  }
807  if (t.is_float() || t.is_bfloat16()) return FloatImm(t, static_cast<double>(value));
808  // For now, we store const scalar values of custom datatypes within doubles; later, during the
809  // datatypes lowering pass, we will lower the value to its true representation in the format
810  // specified by the datatype.
811  // TODO(gus) when do we need to start worrying about doubles not being precise enough?
812  if (static_cast<uint8_t>(t.code()) >= static_cast<uint8_t>(DataType::kCustomBegin)) {
813  return FloatImm(t, static_cast<double>(value));
814  }
815  LOG(FATAL) << "cannot make const for type " << t;
816  return PrimExpr();
817 }
