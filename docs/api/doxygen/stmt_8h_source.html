607  template <typename... Args>

609  Array<Stmt> seq;

620  if (!stmt.defined()) return;
621  if (auto* op = stmt.as<SeqStmtNode>()) {
622  operator()(0, op->seq);
623  } else {
624  seq_->push_back(stmt);
