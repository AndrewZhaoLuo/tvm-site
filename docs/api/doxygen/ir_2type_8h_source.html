409 class FuncTypeNode : public TypeNode {
410  public:
412  Array<Type> arg_types;
414  Type ret_type;
415  // The following fields are used in polymorphic(template) functions
416  // For normal functions, the following two fields will be empty.
418  Array<TypeVar> type_params;
423  Array<TypeConstraint> type_constraints;
424 
425  void VisitAttrs(AttrVisitor* v) {
426  v->Visit("arg_types", &arg_types);
427  v->Visit("ret_type", &ret_type);
428  v->Visit("type_params", &type_params);
429  v->Visit("type_constraints", &type_constraints);
430  v->Visit("span", &span);
431  }
432 
433  bool SEqualReduce(const FuncTypeNode* other, SEqualReducer equal) const {
434  // type params first as they defines type vars.
435  return equal.DefEqual(type_params, other->type_params) && equal(arg_types, other->arg_types) &&
436  equal(ret_type, other->ret_type) && equal(type_constraints, other->type_constraints);
437  }
438 
439  void SHashReduce(SHashReducer hash_reduce) const {
440  hash_reduce.DefHash(type_params);
441  hash_reduce(arg_types);
442  hash_reduce(ret_type);
443  hash_reduce(type_constraints);
444  }
445 
446  static constexpr const char* _type_key = "FuncType";
447  TVM_DECLARE_FINAL_OBJECT_INFO(FuncTypeNode, TypeNode);
