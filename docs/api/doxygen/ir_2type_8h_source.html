402  public:

418  v->Visit("arg_types", &arg_types);

425  bool SEqualReduce(const FuncTypeNode* other, SEqualReducer equal) const {
426  // type params first as they defines type vars.
427  return equal.DefEqual(type_params, other->type_params) && equal(arg_types, other->arg_types) &&
428  equal(ret_type, other->ret_type) && equal(type_constraints, other->type_constraints);
429  }

432  hash_reduce.DefHash(type_params);

434  hash_reduce(ret_type);

436  }
