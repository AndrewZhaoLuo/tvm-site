110 template <typename R, typename... Args>
111 class TracedObjectFunctor {
112  private:
113  using TSelf = TracedObjectFunctor<R, Args...>;
114 
115  template <class TObjectRef, class TCallable>
116  using IsDispatchFunction =
117  typename std::is_convertible<TCallable, std::function<R(TracedObject<TObjectRef>, Args...)>>;
118 
119  public:
131  template <class TObjectRef>
132  R operator()(const String& token, TracedObject<TObjectRef> traced_object, Args... args) const {
133  const runtime::PackedFunc& dispatch_function =
134  GetDispatchFunction(dispatch_table_, token, traced_object.Get()->type_index());
135  return dispatch_function(traced_object.Get(), traced_object.GetPath(), args...);
136  }
137 
147  TSelf& set_dispatch(String token, uint32_t type_index, runtime::PackedFunc f) {
148  SetDispatchFunction(&dispatch_table_, token, type_index, std::move(f));
149  return *this;
150  }
151 
159  template <typename TCallable,
160  typename TObjectRef = typename detail::FirstArgType<TCallable>::ObjectRefType,
161  typename = std::enable_if_t<IsDispatchFunction<TObjectRef, TCallable>::value>>
162  TSelf& set_dispatch(String token, TCallable f) {
163  return set_dispatch(
164  token, //
165  TObjectRef::ContainerType::RuntimeTypeIndex(), //
166  runtime::TypedPackedFunc<R(TObjectRef, ObjectPath, Args...)>(
167  [f = std::move(f)](TObjectRef object, ObjectPath path, Args... args) -> R {
168  return f(MakeTraced(object, path), args...);
169  }));
170  }
180  template <typename TCallable>
181  TSelf& set_dispatch(TCallable&& f) {
182  return set_dispatch(kDefaultDispatchToken, std::forward<TCallable>(f));
183  }
184 
193  void remove_dispatch(String token, uint32_t type_index) {
194  RemoveDispatchFunction(&dispatch_table_, token, type_index);
195  }
196 
197  private:
198  DispatchTable dispatch_table_;
199 };
