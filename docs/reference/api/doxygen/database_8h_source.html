201 class PyDatabaseNode : public DatabaseNode {
202  public:
208  using FHasWorkload = runtime::TypedPackedFunc<bool(const IRModule&)>;
214  using FCommitWorkload = runtime::TypedPackedFunc<Workload(const IRModule&)>;
219  using FCommitTuningRecord = runtime::TypedPackedFunc<void(const TuningRecord&)>;
226  using FGetTopK = runtime::TypedPackedFunc<Array<TuningRecord>(const Workload&, int)>;
231  using FSize = runtime::TypedPackedFunc<int64_t()>;
232 
234  FHasWorkload f_has_workload;
236  FCommitWorkload f_commit_workload;
238  FCommitTuningRecord f_commit_tuning_record;
240  FGetTopK f_get_top_k;
242  FSize f_size;
243 
244  void VisitAttrs(tvm::AttrVisitor* v) {
245  // PackedFuncs are all not visited, because the reflection system doesn't take care of them,
246  // so it cannot be accessible on the python side. If there is such need from the future,
247  // we can then add corresponding accessor methods to help access on python.
248  // `f_has_workload` is not visited
249  // `f_commit_workload` is not visited
250  // `f_commit_tuning_record` is not visited
251  // `f_get_top_k` is not visited
252  // `f_size` is not visited
253  }
254 
255  bool HasWorkload(const IRModule& mod) final {
256  ICHECK(f_has_workload != nullptr) << "PyDatabase's HasWorkload method not implemented!";
257  return f_has_workload(mod);
258  }
259 
260  Workload CommitWorkload(const IRModule& mod) final {
261  ICHECK(f_commit_workload != nullptr) << "PyDatabase's CommitWorkload method not implemented!";
262  return f_commit_workload(mod);
263  }
264 
265  void CommitTuningRecord(const TuningRecord& record) final {
266  ICHECK(f_commit_tuning_record != nullptr)
267  << "PyDatabase's CommitTuningRecord method not implemented!";
268  f_commit_tuning_record(record);
269  }
270 
271  Array<TuningRecord> GetTopK(const Workload& workload, int top_k) final {
272  ICHECK(f_get_top_k != nullptr) << "PyDatabase's GetTopK method not implemented!";
273  return f_get_top_k(workload, top_k);
274  }
275 
276  int64_t Size() final {
277  ICHECK(f_size != nullptr) << "PyDatabase's Size method not implemented!";
278  return f_size();
279  }
280 
281  static constexpr const char* _type_key = "meta_schedule.PyDatabase";
282  TVM_DECLARE_FINAL_OBJECT_INFO(PyDatabaseNode, DatabaseNode);
283 };
